-- DROP FUNCTION public.crear_club(character varying, character varying);

CREATE OR REPLACE FUNCTION public.crear_club(
	p_nombre character varying)
    RETURNS void
    LANGUAGE 'plpgsql'

AS $BODY$
	begin
		insert into fut_clubes(
			nombre, dinero, div_victoria, div_empate, div_derrota, puntos)
		values
        	(p_nombre, 10000, 1, 1, 1, 0);
    end;
$BODY$;

ALTER FUNCTION public.crear_club(character varying, character varying)
    OWNER TO postgres;
