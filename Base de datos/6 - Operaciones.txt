-- drop table fut_operaciones;

CREATE SEQUENCE public.sec_id_operaciones
    INCREMENT 1
    START 1
    MINVALUE 1
    MAXVALUE 99999
    CACHE 1;

ALTER SEQUENCE public.sec_id_ordenes
    OWNER TO postgres;
    
create table public.fut_operaciones (
    operaciones_id bigint default nextval('sec_id_operaciones'),
    tipo char,
    jugador_id bigint,
    club_id bigint,
    cantidad int,
    fecha date,
    precio bigint,
    constraint fk_id_jugador foreign key (jugador_id)
		references public.fut_jugadores(jugador_id) match simple
		on update no action
    	on delete no action,
	constraint fk_id_club foreign key (club_id)
    references public.fut_clubes(club_id) match simple
		on update no action
    	on delete no action
)
with (
    oids = false
);

alter table public.fut_operaciones
    owner to postgres;
